        Command line interface to XTEST

        \begin{columns}
            \begin{column}{5cm}
                \begin{itemize}
                    \item Send input events
                    \item Manipulate windows
                \end{itemize}

                \vskip 1em

                \begin{itemize}
                    \item X
                    \item Same desktop
                \end{itemize}
            \end{column}

            \begin{column}{7cm}
                \begin{semiverbatim}
                    xdotool key ctrl+o

                    sleep 1

                    xdotool type "20yearsofKDE.pdf"

                    sleep 1

                    xdotool key Return
                \end{semiverbatim}
            \end{column}
        \end{columns}
